makeGame =>
  Set up a game, generate game ID

newPlayer (game ID, player name) =>
  Set up player, score

startGame (game ID) =>
  Initialize, shuffle the deck
  Initialize the round number and max round number

startRound (game ID) =>
  Update round number
  Reset tricks
  Shuffle deck
  Update trump
  Deal hands
  Change state to waiting for judgements

judge (game ID, player ID, judgement) =>
  update judgement list
  if all judgements received, change state to waiting for player to put down card
  
playCard (game ID, player ID, card) =>
  verify card is legal
  add card to table
  remove from player's hand
  if first player,
    update chaal
  if last player,
    calculate winner of trick
    add to tricks
    if hand is empty,
      update scores
      if game is not over,
        Start round
      else,
        change state to game over
